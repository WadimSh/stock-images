# Возврат файла по ссылке

 + HTTP Method: GET
 
 + URL: https://stock.werstack.com/images/c41562be-5668-4b0f-88a0-d7b1eafb146d/zateya.jpg

 + Status: 200

 + Headers: 
 ```javascript

  Content-Length: 132057,
  Content-Disposition: inline; zateya.jpg,

 ```

 + Body:
